    <title>عرض رقم عشوائي من JSON</title>
</head>
<body>
    <h1>رقم عشوائي</h1>
    <p id="randomNumberDisplay"></p>
    <button onclick="generateRandomNumber()">عرض رقم عشوائي</button>

    <script>
        function generateRandomNumber() {
            // قراءة محتوى ملف JSON
            fetch('data.json')
                .then(response => response.json())
                .then(data => {
                    // تحديد رقم عشوائي من الأرقام المتاحة
                    const randomNumbers = data.numbers;

                    if (randomNumbers.length > 0) {
                        const randomIndex = Math.floor(Math.random() * randomNumbers.length);
                        const randomNumber = randomNumbers[randomIndex];

                        // عرض الرقم في الصفحة
                        document.getElementById('randomNumberDisplay').textContent = 'الرقم العشوائي: ' + randomNumber;
                    } else {
                        document.getElementById('randomNumberDisplay').textContent = 'لقد تم عرض كل الأرقام.';
                    }
                })
                .catch(error => console.error('حدث خطأ في قراءة الملف JSON:', error));
        }
    </script>
</body>
